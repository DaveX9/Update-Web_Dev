//Pull by hashtag// 
const hashtag = "#MyHashtag"; // Replace with your desired hashtag

async function fetchAlbumsByHashtag() {
  try {
    const response = await fetch("https://graph.facebook.com/v12.0/me/albums?fields=name,description,photos{images}");
    if (!response.ok) {
      throw new Error(`Error fetching albums: ${response.status}`);
    }

    const allAlbums = await response.json();
    const filteredAlbums = allAlbums.data.filter((album) =>
      album.description && album.description.includes(hashtag)
    );

    console.log("Filtered albums by hashtag:", filteredAlbums);
    // Process and display the filtered albums on your page
  } catch (error) {
    console.error("Error fetching albums:", error.message);
  }
}

fetchAlbumsByHashtag();

//Pull By Id//
const albumIds = ["1234567890", "0987654321", "1122334455"]; // Replace with your desired album IDs

async function fetchSpecificAlbums() {
  try {
    const albums = [];
    for (const albumId of albumIds) {
      const response = await fetch(`https://graph.facebook.com/v12.0/${albumId}?fields=name,photos{images}`);
      if (!response.ok) {
        throw new Error(`Error fetching album ${albumId}: ${response.status}`);
      }
      const album = await response.json();
      albums.push(album);
    }
    console.log("Fetched specific albums:", albums);
    // Process and display the albums on your page
  } catch (error) {
    console.error("Error fetching specific albums:", error.message);
  }
}

fetchSpecificAlbums();
